class Vetor {

    var vet[5] ;

    def Vetor(a, b, c, d, e);

    def printaVetor();

}

def Vetor::Vetor(a, b, c, d, e) {

    vet[0] = a ;
    vet[1] = b ;
    vet[2] = c ;
    vet[3] = d ;
    vet[4] = e ;

}

def Vetor::printaVetor(; cont) {

    cont = 0 ;
    while ( cont < 5 ) {
        print(vet[cont], " - ");
    }

}

def main (; vetor ) {
    vetor = Vetor(5,2,3,1,4) ;
    vetor->printaVetor();
}
